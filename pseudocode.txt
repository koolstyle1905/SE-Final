Void buttonLogin_Click
Input : username, password
    If username and password is empty Then
        Display "Please fill in all fields."
        Return
    EndIf
    Try
        check username and password in database
        If not valid username and password Then
            Display "Username or password isn't valid."
            Return
        EndIf
    Catch
        Display "Cannot connect to the server."
        Return
    EndCatch
    EndTry
    Display FormMain
    Delete FormLogin
End buttonLogin_Click

Form CheckExistMdiChildren
Input : formType
    Foreach item in MdiChildren
        If type of item = formType Then
            Return item
        EndIf
    EndForeach
    Return null    
End CheckExistMdiChildren
