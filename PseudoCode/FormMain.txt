Begin FormMain

    Begin void InitializeComponent
        barBtnLogout
        barBtnConfigure
        barBtnExit        
        barBtnRefesh
        navBarStudentDetail
        navBarAdd
        navBarRemove
        navBarRoomDetail
        navBarChangeRoom
        navBarPassword
        navBarContactUs
    End InitializeComponent

    Begin void barBtnLogout_Click
        Delete FormMain
        Display FormLogin
    End barBtnLogout_Click

    Begin void barBtnConfigure_Click
        Display dialog FormConfig
    End barBtnConfigure_Click

    Begin void barBtnExit_Click
        Exit Application
    End barBtnExit_Click

    Begin void barBtnRefesh_Click
        Refesh all opening form
    End barBtnRefesh_Click

    Begin navBarStudentDetail_Click
        If FormStudent is exist Then
            Activate FormStudent
        EndIf
        Else
            Display FormStudent
        EndElse
    End navBarStudentDetail_Click

    Begin navBarAdd_Click
        Display dialog FormAddStudent
    End navBarAdd_Click

    Begin navBarRemove
        Remove the selected student form the database
    End navBarRemove

    Begin navBarRoomDetail_Click
        If FormRoom is exist Then
            Activate FormRoom
        EndIf
        Else
            Display FormRoom
        EndElse
    End navBarRoomDetail_Click

    Begin navBarChangeRoom_Click
       Display dialog FormChangeRoom
    End navBarChangeRoom_Click

    Begin navBarPassword_Click
       Display dialog FormChangePassword
    End navBarPassword_Click

    Begin navBarContactUs_Click
        Display dialog FormContactUs
    End navBarContactUs_Click

End FormMain



Form CheckExistMdiChildren
Input : formType
    Foreach item in MdiChildren
        If type of item = formType Then
            Return item
        EndIf
    EndForeach
    Return null    
End CheckExistMdiChildren
